function nodeSwap(id1, id2) { var elementOne = document.getElementById(String(id1)); var elementTwo = document.getElementById(String(id2)); if (elementOne === null || elementTwo === null) { return; } if (elementOne.contains(elementTwo) || elementTwo.contains(elementOne)) { return; } var elementOneCopy = elementOne.cloneNode(true); var elementTwoCopy = elementTwo.cloneNode(true); elementOne.parentElement.replaceChild(elementTwoCopy, elementOne); elementTwo.parentElement.replaceChild(elementOneCopy, elementTwo); return true; }
